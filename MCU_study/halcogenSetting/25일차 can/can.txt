#define DCNT            8 // data count

void wait(int delay)
{
    int i;
    for(i=0; i<delay; i++)
        ;
}

int main(void)
{
/* USER CODE BEGIN (3) */
    gioInit();
    printf("GIO Init Success\n"); // 속도가 느려짐

    _enable_IRQ_interrupt_();
    canInit();

    printf("CAN Init Success\n");

    canEnableErrorNotification(canREG1);

    for(;;)
    {
        wait(33333333);
        canTransmit(canREG1, canMESSAGE_BOX1, (const uint8 *)&tx_data[0]);
    }
/* USER CODE END */

    return 0;
}


/* USER CODE BEGIN (4) */
void canMessageNotification(canBASE_t *node, uint32_t notification)
{
    if(node == canREG1)
    {
        while(!canIsRxMessageArrived(canREG1, canMESSAGE_BOX2))
            ;

        canGetData(canREG1, canMESSAGE_BOX2, (uint8 *)&rx_data[0]);
        printf("rx_data : %x\n", *rx_data);
    }
}
